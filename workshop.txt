description=[h2]Example Usage[/h2]
description=Use ReactiveX to construct a simple cheer:
description=
description=[code]
description=local rx = require("reactivex")
description=
description=rx.Observable.fromRange(1, 8)
description=  :filter(function(x) return x % 2 == 0 end)
description=  :concat(rx.Observable.of('who do we appreciate'))
description=  :map(function(value) return value .. '!' end)
description=  :subscribe(print)
description=
description=-- => 2! 4! 6! 8! who do we appreciate!
description=[/code]
